npm install tailwindcss lucide-react react-router-dom framer-motion

# Stap 3: Initialiseer Tailwind CSS
npx tailwindcss init -p

# Stap 4: Update je Tailwind configuratie in tailwind.config.js
# (De inhoud hiervan wordt in een apart bestand geleverd)

# Stap 5: Vervang de src-map inhoud met de geleverde code
# (Deze bestanden worden apart geleverd)

npm start

# Stap 7: Voor productie build
npm run build
